% cite papers and books on slides
\DeclareCiteCommand{\shortcite}{}{%

\DeclareCiteCommand{\bookcite}{}{%
\printnames[family]{author}. \emph{\printfield{title}}, \printlist{publisher} \printfield{year}}{;}{}

% cite paper on the bottom of the slide
\renewcommand*\footnoterule{}
\newcommand\blfootnote[1]{%
  \begingroup
  \renewcommand\thefootnote{}\footnote{#1\\\vspace*{-1em}}%
  \addtocounter{footnote}{-1}%
  \endgroup
}
\renewcommand{\footcite}[1]{\blfootnote{\fontsize{7pt}{8pt}\selectfont\shortcite{#1}}}
\newcommand{\footbookcite}[1]{\blfootnote{\fontsize{7pt}{8pt}\selectfont\bookcite{#1}}}

% change 'and' to '&' in listing names
